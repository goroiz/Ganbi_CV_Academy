    <!-- Micro-Interaction Script for Filter Pills -->
    <script>
        function filterSkills(category) {
            // Update filter pill active states
            document.querySelectorAll('.filter-pill').forEach(btn => btn.classList.remove('active'));
            document.getElementById('btn-' + category).classList.add('active');

            // Toggle skills nodes visibility/opacity smoothly
            document.querySelectorAll('.skill-node').forEach(node => {
                if (category === 'all' || node.getAttribute('data-cat') === category) {
                    node.classList.remove('hidden-node');
                } else {
                    node.classList.add('hidden-node');
                }
            });
        }
    </script>
